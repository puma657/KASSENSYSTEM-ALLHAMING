  <script>
    const pages = {
      setup: document.getElementById("setupPage"),
      sales: document.getElementById("salesPage"),
      payment: document.getElementById("paymentPage")
    };

    const pageTitle = document.getElementById("pageTitle");
    const backButton = document.getElementById("backButton");
    const finishButton = document.getElementById("finishButton");
    const productNameInput = document.getElementById("productName");
    const productPriceInput = document.getElementById("productPrice");
    const addProductButton = document.getElementById("addProductButton");
    const productList = document.getElementById("productList");
    const productButtons = document.getElementById("productButtons");
    const cartList = document.getElementById("cartList");
    const clearCartButton = document.getElementById("clearCartButton");
    const moneyButtons = document.getElementById("moneyButtons");
    const paymentTotal = document.getElementById("paymentTotal");
    const paidAmount = document.getElementById("paidAmount");
    const changeAmount = document.getElementById("changeAmount");
    const setupNotice = document.getElementById("setupNotice");
    const paymentNotice = document.getElementById("paymentNotice");

    const moneyOptions = [0.05, 0.10, 0.20, 0.50, 1, 2, 5, 10, 20, 50, 100];
    let currentPage = "setup";
    let products = [];
    let cart = [];
    let paid = 0;

    function euro(value) {
      return value.toLocaleString("de-AT", {
        style: "currency",
        currency: "EUR"
      });
    }

    function showNotice(element, message) {
      element.textContent = message;
      element.classList.add("show");
    }

    function hideNotice(element) {
      element.textContent = "";
      element.classList.remove("show");
    }

    function setPage(page) {
      currentPage = page;
      Object.values(pages).forEach((pageElement) => pageElement.classList.remove("active"));
      pages[page].classList.add("active");

      pageTitle.textContent = {
        setup: "Produkte anlegen",
        sales: "Kassieren",
        payment: "Bezahlung"
      }[page];

      backButton.classList.toggle("visible", page !== "setup");
      finishButton.textContent = page === "payment" && paid >= getCartTotal() ? "Fertig" : "Fertig";

      if (page === "sales") {
        renderSales();
      }

      if (page === "payment") {
        renderPayment();
      }
    }

    function getCartTotal() {
      return cart.reduce((sum, item) => sum + item.price * item.quantity, 0);
    }

    function addProduct() {
      const name = productNameInput.value.trim();
      const price = Number(productPriceInput.value.replace(",", "."));

      if (!name) {
        showNotice(setupNotice, "Bitte gib einen Produktnamen ein.");
        return;
      }

      if (!Number.isFinite(price) || price <= 0) {
        showNotice(setupNotice, "Bitte gib einen gültigen Preis ein.");
        return;
      }

      hideNotice(setupNotice);
      products.push({
        id: Date.now().toString(),
        name,
        price: Math.round(price * 100) / 100
      });

      productNameInput.value = "";
      productPriceInput.value = "";
      productNameInput.focus();
      renderProductList();
    }

    function removeProduct(id) {
      products = products.filter((product) => product.id !== id);
      cart = cart.filter((item) => item.id !== id);
      renderProductList();
    }

    function renderProductList() {
      if (products.length === 0) {
        productList.innerHTML = "<p>Noch keine Produkte angelegt.</p>";
        return;
      }

      productList.innerHTML = "";
      products.forEach((product) => {
        const row = document.createElement("div");
        row.className = "product-row";
        row.innerHTML = `
          <strong>${escapeHtml(product.name)}</strong>
          <span>${euro(product.price)}</span>
        `;

        const deleteButton = document.createElement("button");
        deleteButton.className = "danger";
        deleteButton.type = "button";
        deleteButton.textContent = "Löschen";
        deleteButton.addEventListener("click", () => removeProduct(product.id));

        row.appendChild(deleteButton);
        productList.appendChild(row);
      });
    }

    function renderSales() {
      productButtons.innerHTML = "";
      products.forEach((product) => {
        const button = document.createElement("button");
        button.className = "product-choice";
        button.type = "button";
        button.innerHTML = `
          <span class="product-name">${escapeHtml(product.name)}</span>
          <span class="product-price">${euro(product.price)}</span>
        `;
        button.addEventListener("click", () => addToCart(product));
        productButtons.appendChild(button);
      });

      renderCart();
    }

    function addToCart(product) {
      const existingItem = cart.find((item) => item.id === product.id);

      if (existingItem) {
        existingItem.quantity += 1;
      } else {
        cart.push({ ...product, quantity: 1 });
      }

      renderCart();
    }

    function changeQuantity(id, amount) {
      const item = cart.find((cartItem) => cartItem.id === id);
      if (!item) return;

      item.quantity += amount;
      if (item.quantity <= 0) {
        cart = cart.filter((cartItem) => cartItem.id !== id);
      }

      renderCart();
    }

    function renderCart() {
      if (cart.length === 0) {
        cartList.innerHTML = "<p>Der Einkauf ist noch leer.</p>";
        return;
      }

      cartList.innerHTML = "";
      cart.forEach((item) => {
        const row = document.createElement("div");
        row.className = "cart-row";
        row.innerHTML = `
          <strong>${item.quantity} × ${escapeHtml(item.name)}</strong>
          <span>${euro(item.price * item.quantity)}</span>
        `;

        const controls = document.createElement("div");
        controls.style.display = "flex";
        controls.style.gap = "8px";

        const minus = document.createElement("button");
        minus.className = "light";
        minus.type = "button";
        minus.textContent = "-";
        minus.addEventListener("click", () => changeQuantity(item.id, -1));

        const plus = document.createElement("button");
        plus.className = "light";
        plus.type = "button";
        plus.textContent = "+";
        plus.addEventListener("click", () => changeQuantity(item.id, 1));

        controls.append(minus, plus);
        row.appendChild(controls);
        cartList.appendChild(row);
      });

      const totalRow = document.createElement("div");
      totalRow.className = "cart-row";
      totalRow.innerHTML = `<strong>Gesamt</strong><strong>${euro(getCartTotal())}</strong><span></span>`;
      cartList.appendChild(totalRow);
    }

    function renderPayment() {
      const total = getCartTotal();
      const change = Math.max(0, paid - total);

      paymentTotal.textContent = euro(total);
      paidAmount.textContent = euro(paid);
      changeAmount.textContent = euro(change);

      if (paid > 0 && paid < total) {
        showNotice(paymentNotice, "Der gegebene Betrag reicht noch nicht aus.");
      } else {
        hideNotice(paymentNotice);
      }

      moneyButtons.innerHTML = "";
      moneyOptions.forEach((amount) => {
        const button = document.createElement("button");
        button.className = "money-choice";
        button.type = "button";
        button.textContent = euro(amount);
        button.addEventListener("click", () => {
          paid = Math.round((paid + amount) * 100) / 100;
          renderPayment();
        });
        moneyButtons.appendChild(button);
      });

      const exactButton = document.createElement("button");
      exactButton.className = "money-choice secondary";
      exactButton.type = "button";
      exactButton.textContent = "Passend";
      exactButton.addEventListener("click", () => {
        paid = total;
        renderPayment();
      });

      const resetButton = document.createElement("button");
      resetButton.className = "money-choice danger";
      resetButton.type = "button";
      resetButton.textContent = "Geld löschen";
      resetButton.addEventListener("click", () => {
        paid = 0;
        renderPayment();
      });

      moneyButtons.append(exactButton, resetButton);
    }

    function finish() {
      if (currentPage === "setup") {
        if (products.length === 0) {
          showNotice(setupNotice, "Bitte lege zuerst mindestens ein Produkt an.");
          return;
        }
        hideNotice(setupNotice);
        setPage("sales");
        return;
      }

      if (currentPage === "sales") {
        if (cart.length === 0) {
          return;
        }
        paid = 0;
        setPage("payment");
        return;
      }

      if (currentPage === "payment") {
        if (paid < getCartTotal()) {
          showNotice(paymentNotice, "Bitte wähle genug Geld aus, bevor du abschließt.");
          return;
        }
        cart = [];
        paid = 0;
        setPage("sales");
      }
    }

    function goBack() {
      if (currentPage === "payment") {
        setPage("sales");
      } else if (currentPage === "sales") {
        setPage("setup");
      }
    }

    function escapeHtml(text) {
      const element = document.createElement("span");
      element.textContent = text;
      return element.innerHTML;
    }

    addProductButton.addEventListener("click", addProduct);
    productNameInput.addEventListener("keydown", (event) => {
      if (event.key === "Enter") productPriceInput.focus();
    });
    productPriceInput.addEventListener("keydown", (event) => {
      if (event.key === "Enter") addProduct();
    });
    finishButton.addEventListener("click", finish);
    backButton.addEventListener("click", goBack);
    clearCartButton.addEventListener("click", () => {
      cart = [];
      renderCart();
    });

    renderProductList();
  </script>
